243template <typename T>
244double Matriz<T>::autovalor(const Matriz<T> &other, double tolerancia, int r) const
245{
246 if (colunas_ != linhas_)
247 throw std::invalid_argument("Erro: Matriz deve ser quadrada.");
248
249 if (other.colunas_ != 1 || other.linhas_ != linhas_)
250 throw std::invalid_argument("Erro: O vetor inicial deve ser um vetor "
251 "coluna com a mesma quantidade "
252 "de linhas da matriz.");
253
254 Matriz<T> v = other;
255 Matriz<T> v2(linhas_, 1);
256 Matriz<T> transposta = this->transposta();
257
258 double erro = tolerancia + 1; // Garantir entrada no loop
259 int i = 0;
260
261 while (erro > tolerancia && i < r) {
262 v2 = transposta * v;
263
264 double modulo_v = v.modulo();
265 double modulo_v2 = v2.modulo();
266 erro = std::abs(modulo_v2 - modulo_v);
267
268 v = v2 / modulo_v2; // Normaliza o vetor para evitar crescimento
269 // descontrolado
270 i++;
271 }
272
273 // Cálculo do autovalor final
274 Matriz<T> produto = v.transposta() * (*this) * v;
275 Matriz<T> denominador = v.transposta() * v;
276
277 if (denominador(0, 0) == 0)
278 throw std::runtime_error(
279 "Erro: Denominador zero, impossibilitando o cálculo do autovalor.");
280
281 return produto(0, 0) / denominador(0, 0);
282}

357template <typename T>
358void Matriz<T>::redimensionar(int linhas, int colunas)
359{
360 if (linhas < 0 || colunas < 0)
361 throw std::invalid_argument("Erro: Valores devem ser positivos.");
362
363 const size_t novo_linhas = static_cast<size_t>(linhas);
364 const size_t novo_colunas = static_cast<size_t>(colunas);
365 const size_t novo_tamanho = novo_linhas * novo_colunas;
366
367 if (novo_linhas == linhas_ && novo_colunas == colunas_)
368 return; // Uai?
369
370 // Nós não precisamos criar uma nova matriz nem nada do tipo, basta aumentar
371 // o tamanho do vetor que guarda os dados (matriz), atualizar o índice dos
372 // valores internos e atualizar os valores de linhas e colunas.
373
374 // Nota: fazer resize em um vetor é equivalente a realocá-lo na memória. :)
375
376 const size_t min_linhas = std::min(novo_linhas, linhas_);
377 const size_t min_colunas = std::min(novo_colunas, colunas_);
378
379 // Nós só vamos precisar shiftar valores se o número de COLUNAS mudar.
380 // Se o novo número de colunas for maior, nós precisamos shiftar os valores
381 // para direita começando de trás pra frente, para evitar sobrescrita.
382
383 if (novo_colunas > colunas_) {
384 // Mover linhas existentes de trás pra frente (evitando sobrescrita)
385 matriz_.resize(novo_tamanho, T()); // T() ao invés de 0 para inicializar
386 for (size_t i = min_linhas; i > 0; --i) {
387 const size_t lin = i - 1;
388
389 auto *dest = &matriz_[lin * novo_colunas];
390 auto *src = &matriz_[lin * colunas_];
391 std::memmove(dest, src, min_colunas * sizeof(T));
392
393 // std::memmove não esvazia as posições antigas, precisamos zerar
394 auto *fill_pos = &matriz_[lin * novo_colunas + min_colunas];
395 std::fill_n(fill_pos, novo_colunas - min_colunas, T());
396 }
397 } else if (novo_colunas < colunas_) {
398 // Mover linhas esquerda pra direita, sem risco de sobrescrita
399 for (size_t lin = 0; lin < min_linhas; ++lin) {
400 auto *dest = &matriz_[lin * novo_colunas];
401 auto *src = &matriz_[lin * colunas_];
402 std::memmove(dest, src, min_colunas * sizeof(T));
403 }
404 matriz_.resize(novo_tamanho, T());
405 } else {
406 // Colunas permanecem as mesmas, redimensionar de acordo com as linhas
407 matriz_.resize(novo_tamanho, T());
408 }
409
410 linhas_ = novo_linhas;
411 colunas_ = novo_colunas;
412}

427template <typename T>
428void Matriz<T>::trocar_linhas(int indice1, int indice2)
429{
430 if (indice1 < 0 || indice2 < 0)
431 throw std::out_of_range("Erro: Índice/s de linha negativo/s.");
432
433 if ((size_t)indice1 >= linhas_ || (size_t)indice2 >= linhas_)
434 throw std::out_of_range("Erro: Índice/s de linha além do limite da matriz.");
435
436 auto start1 = matriz_.begin() + indice1 * colunas_; // inicio do intervalo [
437 auto last1 = start1 + colunas_; // fim do intervalo )
438 auto start2 = matriz_.begin() + indice2 * colunas_; // inicio do destino
439 std::swap_ranges(start1, last1, start2);
440}
