28Matriz<T> sub(Matriz<T> &matriz, Matriz<T> &b)
29{
30 int n = static_cast<int>(matriz.linhas());
31 Matriz<T> x(n, 1);
32
33 for (int i = n - 1; i >= 0; i--) {
34 T soma = 0;
35 for (int j = i + 1; j < n; j++) {
36 soma += matriz(i, j) * x(j, 0);

39 }

105template <typename T>
106Matriz<T> gauss(Matriz<T> &matriz, Matriz<T> &b)
107{
108 if (!verificar_matriz_quadrada(matriz)) {
109 throw std::invalid_argument("Erro: Matriz não é quadrada.");
110 };
111 int n = static_cast<int>(matriz.linhas());
112
113 for (int i = 0; i < n - 1; ++i) {
114 for (int k = i + 1; k < n; ++k) {
115 T fator = matriz(k, i) / matriz(i, i);
116
117 for (int j = i; j < n; ++j) {
118 matriz(k, j) -= matriz(i, j) * fator;
119 }
120
121 b(k, 0) -= b(i, 0) * fator;
122 }
123 }
124
125 return matriz;

139template <typename T>
140Matriz<T> gauss_jordan(Matriz<T> &matriz, Matriz<T> &b)
141{
142 if (!verificar_matriz_quadrada(matriz)) {
143 throw std::invalid_argument("Erro: Matriz não é quadrada.");
144 };
145 int n = static_cast<int>(matriz.linhas());
146
147 // Aplicação do escalonamento direto (Gauss)
148 gauss(matriz, b);
149
150 // Aplicação do escalonamento reverso (Jordan)
151 for (int i = n - 1; i >= 0; --i) {
152 T divisor = matriz(i, i);
153 for (int j = 0; j < n; ++j) {
154 matriz(i, j) /= divisor;
155 }
156 b(i, 0) /= divisor;
157
158 for (int k = i - 1; k >= 0; --k) {
159 T fator = matriz(k, i);
160 for (int j = 0; j < n; ++j) {
161 matriz(k, j) -= matriz(i, j) * fator;
162 }
163 b(k, 0) -= b(i, 0) * fator;
164 }
165 }
166
167 return matriz;
168}

214template <typename T>
215Matriz<T> retorna_L(Matriz<T> &matriz)
216{
217 if (!verificar_matriz_quadrada(matriz)) {
218 throw std::invalid_argument("Erro: Matriz não é quadrada.");
219 }
220
221 int n = static_cast<int>(matriz.linhas());
222 Matriz<T> L(n, n); // Criamos uma nova matriz L
223
224 // Inicializa a matriz L com a identidade (valores 1 na diagonal e 0 fora da diagonal)
225 for (int i = 0; i < n; ++i) {
226 L(i, i) = 1; // Diagonal principal é 1
227 for (int j = i + 1; j < n; ++j) {
228 L(i, j) = 0; // Valores acima da diagonal são 0
229 }
230 }
231
232 for (int i = 0; i < n - 1; ++i) {
233 for (int k = i + 1; k < n; ++k) {
234 T fator = matriz(k, i) / matriz(i, i);
235 L(k, i) = fator; // Armazena o fator na matriz L
236
237 for (int j = i; j < n; ++j) {
238 // Atualiza a matriz original (ou já modificada) para a próxima iteração
239 matriz(k, j) -= matriz(i, j) * fator;
240 }
241 }
242 }
243
244 return L; // Retorna a matriz L
245}
